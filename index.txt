      <Card className="w-full max-w-md">

        </CardHeader>

          </p>

            <Button 
              onClick={handleClick}
              className="w-full"
              variant="default"
            >
              Click me! 
            </Button>
            
            {clickCount > 0 && (
              <div className="bg-muted p-3 rounded-md">
                <p className="text-sm text-muted-foreground">
                  Button clicked: <span className="font-semibold text-primary">{clickCount}</span> time{clickCount !== 1 ? 's' : ''}
                </p>
